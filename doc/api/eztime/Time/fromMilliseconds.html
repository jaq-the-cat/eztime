static Time fromMilliseconds(int ms) {
  double h = _msToHours(ms).abs();
  int justH = h.truncate();
  double m = (h - justH) * 60;
  int justM = m.truncate();
  double s = (m - justM) * 60;

  return Time(justH, justM, s, ms < 0);
}
